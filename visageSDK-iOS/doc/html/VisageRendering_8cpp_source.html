26 void gluLookAt(GLfloat eyex, GLfloat eyey, GLfloat eyez,
27  GLfloat centerx, GLfloat centery, GLfloat centerz,
28  GLfloat upx, GLfloat upy, GLfloat upz)
29 {
30  GLfloat m[16];
31  GLfloat x[3], y[3], z[3];
32  GLfloat mag;
33  GLfloat invmag;
34 
35  /* Make rotation matrix */
36 
37  /* Z vector */
38  z[0] = eyex - centerx;
39  z[1] = eyey - centery;
40  z[2] = eyez - centerz;
41  mag = sqrt(z[0] * z[0] + z[1] * z[1] + z[2] * z[2]);
42  if (mag) { /* mpichler, 19950515 */
43  invmag = 1.0f / mag;
44  z[0] *= invmag;
45  z[1] *= invmag;
46  z[2] *= invmag;
47  }
48 
49  /* Y vector */
50  y[0] = upx;
51  y[1] = upy;
52  y[2] = upz;
53 
54  /* X vector = Y cross Z */
55  x[0] = y[1] * z[2] - y[2] * z[1];
56  x[1] = -y[0] * z[2] + y[2] * z[0];
57  x[2] = y[0] * z[1] - y[1] * z[0];
58 
59  /* Recompute Y = Z cross X */
60  y[0] = z[1] * x[2] - z[2] * x[1];
61  y[1] = -z[0] * x[2] + z[2] * x[0];
62  y[2] = z[0] * x[1] - z[1] * x[0];
63 
64  /* mpichler, 19950515 */
65  /* cross product gives area of parallelogram, which is < 1.0 for
66  * non-perpendicular unit-length vectors; so normalize x, y here
67  */
68 
69  mag = sqrt(x[0] * x[0] + x[1] * x[1] + x[2] * x[2]);
70  if (mag) {
71  invmag = 1.0f / mag;
72  x[0] *= invmag;
73  x[1] *= invmag;
74  x[2] *= invmag;
75  }
76 
77  mag = sqrt(y[0] * y[0] + y[1] * y[1] + y[2] * y[2]);
78  if (mag) {
79  invmag = 1.0f / mag;
80  y[0] *= invmag;
81  y[1] *= invmag;
82  y[2] *= invmag;
83  }
84 
85 #define M(row,col) m[col*4+row]
86  M(0, 0) = x[0];
87  M(0, 1) = x[1];
88  M(0, 2) = x[2];
89  M(0, 3) = 0.0;
90  M(1, 0) = y[0];
91  M(1, 1) = y[1];
92  M(1, 2) = y[2];
93  M(1, 3) = 0.0;
94  M(2, 0) = z[0];
95  M(2, 1) = z[1];
96  M(2, 2) = z[2];
97  M(2, 3) = 0.0;
98  M(3, 0) = 0.0;
99  M(3, 1) = 0.0;
100  M(3, 2) = 0.0;
101  M(3, 3) = 1.0;
102 #undef M
103  glMultMatrixf(m);
104 
105  /* Translate Eye to Origin */
106  glTranslatef(-eyex, -eyey, -eyez);
107 }

148 static void InitNonuniformCatmullRom(float x0, float x1, float x2, float x3, float dt0, float dt1, float dt2, CubicPoly &p)
149 {
150  float t1 = (x1 - x0) / dt0 - (x2 - x0) / (dt0 + dt1) + (x2 - x1) / dt1;
151  float t2 = (x2 - x1) / dt1 - (x3 - x1) / (dt1 + dt2) + (x3 - x2) / dt2;
152 
153  t1 *= dt1;
154  t2 *= dt1;
155 
156  InitCubicPoly(x1, x2, t1, t2, p);
157 }

166 static void InitCentripetalCR(const Vec2D& p0, const Vec2D& p1, const Vec2D& p2, const Vec2D& p3, CubicPoly &px, CubicPoly &py)
167 {
168  float dt0 = powf(VecDistSquared(p0, p1), 0.25f);
169  float dt1 = powf(VecDistSquared(p1, p2), 0.25f);
170  float dt2 = powf(VecDistSquared(p2, p3), 0.25f);
171 
172  if (dt1 < 1e-4f) dt1 = 1.0f;
173  if (dt0 < 1e-4f) dt0 = dt1;
174  if (dt2 < 1e-4f) dt2 = dt1;
175 
176  InitNonuniformCatmullRom(p0.x, p1.x, p2.x, p3.x, dt0, dt1, dt2, px);
177  InitNonuniformCatmullRom(p0.y, p1.y, p2.y, p3.y, dt0, dt1, dt2, py);
178 }

180 static void SetupCamera(int width, int height, float f)
181 {
182  GLfloat x_offset = 1;
183  GLfloat y_offset = 1;
184  if (width > height)
185  x_offset = ((GLfloat)width)/((GLfloat)height);
186  else if (width < height)
187  y_offset = ((GLfloat)height)/((GLfloat)width);
188 
189  //Note:
190  // FOV in radians is: fov*0.5 = arctan ((top-bottom)*0.5 / near)
191  // In this case: FOV = 2 * arctan(frustum_y / frustum_near)
192  //set frustum specs
193  GLfloat frustum_near = 0.001f;
194  GLfloat frustum_far = 30; //hard to estimate face too far away
195  GLfloat frustum_x = x_offset*frustum_near/f;
196  GLfloat frustum_y = y_offset*frustum_near/f;
197  //set frustum
198  glMatrixMode(GL_PROJECTION);
199  glLoadIdentity();
200  #if defined(WIN32) || defined(LINUX) || defined(MAC_OS_X)
201  glFrustum(-frustum_x,frustum_x,-frustum_y,frustum_y,frustum_near,frustum_far);
202  #else
203  glFrustumf(-frustum_x,frustum_x,-frustum_y,frustum_y,frustum_near,frustum_far);
204  #endif
205  glMatrixMode(GL_MODELVIEW);
206  //clear matrix
207  glLoadIdentity();
208  //camera in (0,0,0) looking at (0,0,1) up vector (0,1,0)
209  gluLookAt(0,0,0,0,0,1,0,1,0);
210 
211 }

253  }

257  glDisableClientState(GL_VERTEX_ARRAY);
258 
259  //clean-up
260  delete[] vertices;

400 static int NearestPow2(int n)
401 {
402  unsigned int v; // compute the next highest power of 2 of 32-bit v
403 
404  v = n;
405  v--;
406  v |= v >> 1;
407  v |= v >> 2;
408  v |= v >> 4;
409  v |= v >> 8;
410  v |= v >> 16;
411  v++;
412 
413  return v;
414 }

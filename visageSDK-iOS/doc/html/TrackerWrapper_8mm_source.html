375 //set the camera and its view (frustum)
376 + (void) setupFrustumWithWidth: (int) width //width of image to be drawn
377  andHeight: (int) height //height of image to be drawn
378  andFocus: (float) f //focus
379 {
380  GLfloat x_offset = 1;
381  GLfloat y_offset = 1;
382  if (width > height)
383  x_offset = ((GLfloat)width)/((GLfloat)height);
384  else if (width < height)
385  y_offset = ((GLfloat)height)/((GLfloat)width);
386  //Note:
387  // FOV in radians is: fov*0.5 = arctan ((top-bottom)*0.5 / near)
388  // In this case: FOV = 2 * arctan(frustum_y / frustum_near)
389  //set frustum specs
390  GLfloat frustum_near = 0.001f;
391  GLfloat frustum_far = 30; //hard to estimate face too far away
392  GLfloat frustum_x = x_offset*frustum_near/f;
393  GLfloat frustum_y = y_offset*frustum_near/f;
394  //set frustum
395  glMatrixMode(GL_PROJECTION);
396  glLoadIdentity();
397 
398  glFrustumf(-frustum_x,frustum_x,-frustum_y,frustum_y,frustum_near,frustum_far);
399 
400  glMatrixMode(GL_MODELVIEW);
401  //clear matrix
402  glLoadIdentity();
403  //camera in (0,0,0) looking at (0,0,1) up vector (0,1,0)
404  gluLookAt(0,0,0,0,0,1,0,1,0);
405 }
